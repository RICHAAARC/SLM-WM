+def encode_with_exact_clean_base(encoder, payload, key_material):
+    """重放 encode 的首个高斯采样, 且不改变官方编码器的随机状态演化."""

+    random_state_before = torch.random.get_rng_state()

+        torch.random.set_rng_state(random_state_after)
